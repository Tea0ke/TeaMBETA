let cacheReady = false;

 

/* ===== ЗАГРУЗКА РЕСУРСОВ ===== */

async function loadResources(){

 

let bar=document.getElementById("bar");

let progress=0;

 

// список файлов

const files=[

"https://cdn.jsdelivr.net/npm/mathjs@14/lib/browser/math.js"

];

 

for(let i=0;i<files.length;i++){

 

await fetch(files[i]); // скачали

 

progress += 100/files.length;

bar.style.width = progress+"%";

 

}

 

cacheReady = true;

}

 

/* ===== START ===== */

window.onload=async ()=>{

 

let already = localStorage.getItem("team_cached");

 

if(!already){

 

await loadResources();

localStorage.setItem("team_cached","1");

 

}

 

// быстрый старт

setTimeout(()=>{

document.getElementById("loader").style.display="none";

document.getElementById("app").style.display="block";

},500);

 

};
